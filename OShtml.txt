        (function () {
            var ALLOW_LIGHT = ["sharp", "cool", "sepia", "soft"];
            try {
                var dark = localStorage.getItem("pbpm_dark_mode") === "1";
                var light = localStorage.getItem("pbpm_light_theme") || "sharp";
                if (ALLOW_LIGHT.indexOf(light) === -1) light = "sharp";
                document.documentElement.setAttribute("data-theme", dark ? "dark" : light);
            } catch (e) {
                try {
                    var d2 = localStorage.getItem("pbpm_dark_mode") === "1";
                    document.documentElement.setAttribute("data-theme", d2 ? "dark" : "sharp");
                } catch (e2) {
                    document.documentElement.setAttribute("data-theme", "sharp");
                }
            }
        })();
    </script>
